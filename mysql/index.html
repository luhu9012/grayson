检索练习

– 查看数据库所有表
SHOW TABLES;
– 查看表中所有列
SHOW COLUMNS FROM customers;

– 检索
SELECT * FROM customers;
– 检索表中某一列数据
SELECT cust_id FROM customers;
– 检索多列
SELECT cust_id,cust_name FROM customers;
– 检索不同的值
SELECT vend_id FROM products;
SELECT DISTINCT vend_id FROM products;
– distinct 作用所有列，不仅仅是一列
SELECT vend_id,prod_price FROM products;
SELECT DISTINCT vend_id,prod_price FROM products;

– 检索限制行数 不同dbms实现不一样 top(SQL serve) ROWNUM(oracle)
SELECT prod_name FROM products LIMIT 3;
– 指定从第三行开始检索不超过3条的记录，从0行开始
SELECT prod_name FROM products LIMIT 3 OFFSET 3;
– 检索指定排序顺序 prod_price prod_name
– ORDER BY 必须在句尾，否则报错！！
SELECT * FROM products ORDER BY prod_price,prod_name;
– 指定排序方向 DESC 降序
SELECT * FROM products ORDER BY prod_price DESC;
SELECT * FROM products ORDER BY prod_price DESC,prod_name ASC;

– 过滤数据 where
SELECT * FROM products WHERE prod_price=3.49;
– 操作符： 11个，有些冗余的，is null, between,!= 或 <>, !<或 >=, <=或!>,=

– 高级数据过滤

– 逻辑操作符 and or in not,可以用来租户where子句,其中 and操作符优先级高于or
/*
in 操作符用于指定范围，优点如下：

  1. 在有很多合法选项时，in更清楚更直观
  2. 在与其他and or操作符一起使用in时，求值顺序更容易管理
  3. in 比or 执行得更快
  4. in最大的优点是可以包含其他select语句，能够更动态的建立where子句

*/
SELECT * FROM products WHERE vend_id=’DLL01’ AND prod_price<10;

– 通配符过滤 %，%搜索根据dbms配置区分大小写的,’Fish%’与’fish%’不一样,%代表搜索模式中给定位置的0个、1个或者多个字符。like是谓词不是操作符
SELECT prod_name,prod_id FROM products WHERE prod_name LIKE ‘Fish%’;

– 通配符下划线，与%类似，但_只表示一个字符，DB2不支持

– 通配符方括号[],用来指定一个字符集，必须匹配指定位置的一个字符

/* 使用通配符的注意事项

  1. 尽量不用通配符，使用其他替代方法
  2. 确实需要通配符时，不要用在搜索模式的开始处，这样搜索起来是最慢的
  3. 仔细注意通配符位置，放错地方将会返回不一样的数据

*/

– 创建计算字段
– 拼接不同列，不同dbms有不同实现，如+ || 和concat
– rtrim 去掉列右边空格，trim去掉列数据两边的空格
– as 为列别名，通常是可选的，方便客户端引用
SELECT CONCAT(RTRIM(vend_name),’(‘,TRIM(vend_country),’)’) AS vend_title FROM vendors ORDER BY vend_name DESC;

– 支持+ - * /的运算
SELECT prod_id,quantity,item_price,quantity*item_price AS expand_price FROM orderitems WHERE order_num = 20008;

– 汇总数据 AVG COUNT MAX MIN SUM 函数

– # 分组数据
– GROUP BY和 having
